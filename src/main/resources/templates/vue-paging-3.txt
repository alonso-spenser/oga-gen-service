<template>
  <oga-page :offset="240" :percentage="100" v-loading="state.loading">
    <template #header>
      <div class="oga-filter">
        <div class="oga-filter-item">
          <oga-input
            :label="t('placeholder.label')"
            :placeholder="t('placeholder.search')"
            clearable
            v-model="pageQueryState.params.q"
            @clear="clearSearchCondition"
          >
            <template #append>
              <el-button :loading="state.loading" @click="filterData">
                <oga-icon name="search"></oga-icon>
              </el-button>
            </template>
          </oga-input>
        </div>
        <div class="oga-filter-item">
          <el-button class="el-material-button" type="text" @click="clearSearchCondition">
            <oga-icon name="refresh"></oga-icon>
          </el-button>
          <el-button class="el-material-button ml-5" type="text" @click="add[(${table.config.fun})]Drawer">
            <oga-icon name="plus"></oga-icon>
          </el-button>
        </div>
        <div class="oga-filter-item text-count">
          <small class="text-secondary mr-2">
            {{ t("total") }}
          </small>
          <small class="text-primary">
            {{ pageResult.total }}
          </small>
        </div>
        <div class="oga-filter-item"></div>
      </div>
    </template>
    <oga-table v-model="pageResult" v-bind="paginationState" @paging="getData">
    </oga-table>
  </oga-page>
</template>

<script setup lang="tsx">
import { nextTick, ref, watch } from "vue";
import type { FormInstance, FormRules, TableColumnCtx } from "element-plus";
import { isEmpty } from "@/plugins/utility";
import { formValidation } from "@/plugins/form";
import { usePageState } from "@/plugins/page/page-state";
import { ActionType, ColumnType } from "@/stores/type/page-type";
import i18n from "@/plugins/i18n/base";
import {
  type [# th:text="${table.className}" /]Result,
  type [# th:text="${table.className}" /]Request,
  use[# th:text="${table.className}" /]Store,
} from "@/stores/page/[(${table.config.routerName})]-store";
import {
  fetch[(${table.config.prefix})]Detail,
  fetch[(${table.config.prefix})]Update,
  fetch[(${table.config.prefix})]Paging,
  fetch[(${table.config.prefix})]Delete
} from '@/plugins/api/[(${table.config.catalog})]'

/**
 * i18n
 */
const t = i18n.global.t;

/**
 * Page state
 */
const { state, pageState, pageQueryState, pageResult, paginationState } =
  usePageState<[# th:text="${table.className}" /]Result>();

/**
 * Search parameters
 */
// pageQueryState.params = {
//   ...pageQueryState.params,
//   state: null,
// }

/**
 * [(${table.config.fun})] Store
 */
const [# th:text="${table.lowerClassName}" /]Store = use[# th:text="${table.className}" /]Store();

/**
 * Table configuration
 */
pageState.updatePaginationState({
  actionList: {
    rowClick: (row: [# th:text="${table.className}" /]Result, column: TableColumnCtx) => {
      columnEvents(row, column);
    },
    delete: {
      icon: "",
      label: t("delete.drop"),
      onClick: (rows: Array<[# th:text="${table.className}" /]Result>) => {
        delete[(${table.config.fun})](rows);
      },
    },
  },

  columnList: [[# th:each="item,stat: ${table.children}"]
    [# th:if="${item.primary == 0}"]{
      prop: '[# th:text="${item.javaName}" /]',
      type: ColumnType.Default,
      label: t('[(${table.config.catalog})].[(${table.config.i18n})].tableHeader.[# th:text="${item.javaName}" /]')
    }[# th:if="${not stat.last}" th:text="','" /][/]
  [/]],

  empty: {
    content: t("[(${table.config.catalog})].[(${table.config.i18n})].empty.content"),
    buttonLabel: t("[(${table.config.catalog})].[(${table.config.i18n})].empty.buttonLabel"),
    onClick: () => {
      add[(${table.config.fun})]()
    },
  },
});

/**
 * Load cached query state
 */
pageState.loadCache([# th:text="${table.lowerClassName}" /]Store.pageQueryState, [# th:text="${table.lowerClassName}" /]Store.normalQueryState);

/**
 * Add
 */
const add[(${table.config.fun})] = () => {
  pageState.redirect('/[(${table.config.catalog})]/[(${table.config.path})]/add')
};

/**
 * Update
 */
const update[(${table.config.fun})] = (row: [# th:text="${table.className}" /]Result) => {
  pageState.redirect(`/[(${table.config.catalog})]/[(${table.config.path})]/update/${row.id}`)
};

/**
 * Clear search condition
 */
const clearSearchCondition = () => {
  pageQueryState.current = 1;
  pageQueryState.params = {
    ...pageQueryState.params,
    q : "",
  }
  [# th:text="${table.lowerClassName}" /]Store.updatePageQueryState({
    current: 1,
    params: {
      ...pageQueryState.params,
      q: "",
    }
  });

  getData();
};

/**
 * Filter data
 */
const filterData = () => {
  getData();
};

/**
 * Get paging data
 */
const getData = async () => {
  const { current, size } = pageResult.value;

  [# th:text="${table.lowerClassName}" /]Store.updatePageQueryState({
    current,
    size,
  });

  await pageState.resolveResponse(
    fetch[# th:text="${table.className}" /]Paging({
      ...pageQueryState,
      current,
      size,
    }),
  );
};

/**
 * Table row click
 */
const columnEvents = (row: [# th:text="${table.className}" /]Result, column: TableColumnCtx) => {
  update[(${table.config.fun})](row);
};

/**
 * Delete
 */
const delete[(${table.config.fun})] = async (rows: [# th:text="${table.className}" /]Result[]) => {
  const ids = rows.map((row) => row.id);
  await pageState.confirmDelete(async () => {
    const response = await pageState.submitRequest<[# th:text="${table.className}" /]Request>(
      fetch[# th:text="${table.className}" /]Delete({ ids }),
      ActionType.Delete,
    );
    if (response.success) {
      await getData();
    }
  }, ids.length);
};

/**
 * Search criteria changed;
 * reloading data.
 */
watch(
    () => pageQueryState,
    () => {
      getData()
    },
    { deep: true },
);

/**
 * Initial load
 */
getData();
</script>
