<template>
  <oga-page :offset="240" :percentage="100" v-loading="state.loading">
    <el-form
      ref="formInstance"
      :model="formModel"
      :rules="formRules"
      class="oga-form"
    >
      <el-row :gutter="20" class="el-form-item">
        <el-col :span="12">
        [# th:each="item,stat: ${table.children}"][# th:if="${item.primary == 0 && stat.index % 2 == 0}"]
          <el-form-item prop="[(${item.javaName})]">
            <oga-input
              v-model="formModel.[(${item.javaName})]"
              :label="t('[(${table.config.catalog})].[(${table.config.i18n})].entity.[(${item.javaName})].label')"
              :placeholder="t('[(${table.config.catalog})].[(${table.config.i18n})].entity.[(${item.javaName})].placeholder')"
            ></oga-input>
          </el-form-item>
        [/][/]
        </el-col>
        <el-col :span="12">
          [# th:each="item,stat: ${table.children}"][# th:if="${item.primary == 0 && stat.index % 2 == 1}"]
            <el-form-item prop="[(${item.javaName})]">
              <oga-input
                v-model="formModel.[(${item.javaName})]"
                :label="t('[(${table.config.catalog})].[(${table.config.i18n})].entity.[(${item.javaName})].label')"
                :placeholder="t('[(${table.config.catalog})].[(${table.config.i18n})].entity.[(${item.javaName})].placeholder')"
              ></oga-input>
            </el-form-item>
          [/][/]
        </el-col>
      </el-row>
    </el-form>
    <oga-unsaved
      round
      offset="60%"
      v-model="state.unsaved"
      :loading="state.submitting"
      @cancel="cancelUpdate"
      @confirmed="formSubmit(formInstance)"
    >
    </oga-unsaved>
  </oga-page>
</template>

<script setup lang="tsx">
import { nextTick, ref, watch } from "vue";
import type { FormInstance, FormRules } from "element-plus";
import { isEmpty } from "@/plugins/utility";
import { formValidation } from "@/plugins/form";
import { usePageState } from "@/plugins/page/page-state";
import { ActionType, ColumnType } from "@/stores/type/page-type";
import i18n from "@/plugins/i18n/base";
import {
  type [# th:text="${table.className}" /]Result,
  type [# th:text="${table.className}" /]Request,
  use[# th:text="${table.className}" /]Store,
} from "@/stores/page/[(${table.config.routerName})]-store";
import {
  fetch[(${table.config.prefix})]Detail,
  fetch[(${table.config.prefix})]Update,
  fetch[(${table.config.prefix})]Paging,
  fetch[(${table.config.prefix})]Delete
} from '@/plugins/api/[(${table.config.catalog})]'

/**
 * i18n
 */
const t = i18n.global.t;

/**
 * Page state
 */
const { state, pageState, pageQueryState, pageResult, paginationState } =
  usePageState<[# th:text="${table.className}" /]Result>();

/**
 * [(${table.config.fun})] Store
 */
const [# th:text="${table.lowerClassName}" /]Store = use[# th:text="${table.className}" /]Store();

/**
 * Form instance
 */
const formInstance = ref<FormInstance>();

/**
 * Form model
 */
const formModel = ref<[# th:text="${table.className}" /]Request>([# th:text="${table.lowerClassName}" /]Store.getDefaultModel());

/**
 * Form rules
 */
const formRules = ref<FormRules<[# th:text="${table.className}" /]Request>>({
[# th:each="item,stat: ${table.children}"][# th:if="${item.primary == 0}"]
[(${item.javaName})]: [
  {
    required: true,
    message: t('[(${table.config.catalog})].[(${table.config.i18n})].entity.[(${item.javaName})].required'),
    trigger: 'blur'
  }
][# th:if="${not stat.last}" th:text="','" /][/][/]
});

/**
 * Cancel update
 */
const cancelUpdate = () => {
  pageState.redirect('/[(${table.config.catalog})]/[(${table.config.path})]');
};

/**
 * Get detail
 */
const getDetail = async () => {
  if (isEmpty(pageState.getPrimaryId())) return;

  const data = await pageState.resolveResponse<[# th:text="${table.className}" /]Result>(
    fetch[# th:text="${table.className}" /]Detail({ id: pageState.getPrimaryId() }),
  );

  if (data) {
    formModel.value = { ...data };

    await nextTick(() => {
      state.value.unsaved = false;
    });
  }
};

/**
 * Form submit
 */
const formSubmit = (formEl?: FormInstance) => {
  if (!formEl) return;

  formValidation(formEl, () => {
    update[(${table.config.fun})]();
  });
};

/**
 * Update account
 */
const update[(${table.config.fun})] = async () => {
  const response = await pageState.submitRequest<[# th:text="${table.className}" /]Request>(
    fetch[# th:text="${table.className}" /]Update(formModel.value),
    formModel.value.id ? ActionType.Update : ActionType.Insert,
  );

  if (response.success) {
    cancelUpdate();
  }
};

/**
 * Unsaved detection
 */
watch(
  () => formModel.value,
  () => {
    state.value.unsaved = true;
  },
  { deep: true },
);

/**
 * Initial load
 */
getData();
</script>
